EU Cookie law plugin

This is cookie opt in-out library as per the new EU law released on the 26th May.

Change cookie settings of this page

Usage

Include cookie.debug.min.js and cookie_style.css on your page.

// Initialize widget
EU.CookieManager.init();

Options

// Remove options to use defaults
EU.CookieManager.init({
    idle: 0,
    level: 0,
    message: '[INTRO TEXT]',
    cancel: 'I agree',
    changeSettings: 'Change settings',
    linkText: 'Find out more about cookies',
    functionalList: {
        'strict': {
            will: ['[USE CASE 1]', '[USE CASE 2]'],
            willnot: []
        },
        'functional': {
            will: [],
            willnot: []
        },
        'targeting': {
            will: [],
            willnot: []
        }
      }
});
